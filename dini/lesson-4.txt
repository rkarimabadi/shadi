        questions = questionBank.map(q => {
            let type = currentMode;
            if (currentMode === 'mixed') type = Math.random() > 0.5 ? 'choice' : 'desc';
            return { ...q, type };
        });
        states = questions.map(() => ({ ans: null, text: '', revealed: false, isCorrect: null }));
        currentIdx = 0;
        showSection('quiz');
        renderQuestion();
    }

    function renderQuestion() {
        const q = questions[currentIdx];
        const s = states[currentIdx];
        const card = document.getElementById('question-card');
        document.getElementById('progress').innerText = `سوال ${currentIdx + 1} از ${questions.length}`;

        let html = `<p style="font-size:1.15rem; font-weight:bold; margin-bottom:20px; color:var(--primary); line-height:1.6;">${q.q}</p>`;

        if (q.type === 'choice') {
            q.opts.forEach((o, i) => {
                let cls = 'option-item';
                if (s.ans === i) cls += ' selected';
                if (s.revealed) {
                    if (i === q.c) cls += ' correct';
                    else if (s.ans === i) cls += ' wrong';
                }
                html += `<div class="${cls}" onclick="handleChoice(${i})">${o}</div>`;
            });
            html += `<button class="btn btn-outline" style="margin-top:10px; width:100%" onclick="reveal()">بررسی پاسخ درست</button>`;
        } else {
            html += `<textarea id="q-text" placeholder="پاسخ خود را اینجا بنویسید..." oninput="states[currentIdx].text=this.value">${s.text}</textarea>`;
            html += `<button class="btn btn-outline" style="width:100%" onclick="reveal()">نمایش پاسخ مرجع و خودسنجی</button>`;
        }

        html += `
            <div class="feedback-box ${s.revealed ? 'visible' : ''}">
                <p style="color:var(--secondary); font-size:0.9rem; margin-bottom:5px;">پاسخ دقیق متن کتاب:</p>
                <p style="font-weight:bold; color:var(--primary);">${q.type === 'choice' ? q.opts[q.c] : q.a}</p>
                ${q.type === 'desc' ? `
                    <div style="display:flex; gap:10px; margin-top:10px">
                        <button class="btn btn-outline" style="flex:1; border-color:var(--success); color:var(--success)" onclick="setEval(true)">درست نوشتم <i class="ph ph-check"></i></button>
                        <button class="btn btn-outline" style="flex:1; border-color:var(--error); color:var(--error)" onclick="setEval(false)">اشتباه نوشتم <i class="ph ph-x"></i></button>
                    </div>
                ` : ''}
            </div>
        `;
        card.innerHTML = html;
        updateNav();
    }

    function handleChoice(i) {
        if (states[currentIdx].revealed) return;
        states[currentIdx].ans = i;
        states[currentIdx].isCorrect = (i === questions[currentIdx].c);
        renderQuestion();
    }

    function reveal() {
        states[currentIdx].revealed = true;
        renderQuestion();
    }

    function setEval(v) {
        states[currentIdx].isCorrect = v;
        renderQuestion();
    }

    function nav(d) {
        if (d === 1 && currentIdx === questions.length - 1) { finish(); return; }
        currentIdx += d;
        renderQuestion();
    }

    function updateNav() {
        document.getElementById('prev-btn').disabled = currentIdx === 0;
        document.getElementById('next-btn').innerHTML = currentIdx === questions.length - 1 ? 'مشاهده نتیجه' : 'سوال بعدی <i class="ph ph-arrow-left"></i>';
    }

    function finish() {
        showSection('result');
        const score = states.filter(s => s.isCorrect === true).length;
        document.getElementById('final-score').innerText = `${score} از ${questions.length}`;
        document.getElementById('result-text').innerText = score > 25 ? "آفرین! شما برای عبوری آسان آماده هستید." : "پیشنهاد می‌شود یک بار دیگر درسنامه را با دقت مطالعه کنید.";
        saveToStorage(score, questions.length);
    }

    function saveToStorage(s, t) {
        const logs = JSON.parse(localStorage.getItem('payam_logs_lesson4') || '[]');
        logs.push({ date: new Date().toLocaleDateString('fa-IR'), score: `${s}/${t}` });
        localStorage.setItem('payam_logs_lesson4', JSON.stringify(logs.slice(-5)));
    }

    function renderHistory() {
        const logs = JSON.parse(localStorage.getItem('payam_logs_lesson4') || '[]');
        const section = document.getElementById('history-section');
        const list = document.getElementById('history-list');
        if (logs.length > 0) {
            section.classList.remove('hidden');
            list.innerHTML = logs.reverse().map(l => `
                <div class="history-card">
                    <div><div style="font-size:0.8rem; color:gray">${l.date}</div></div>
                    <div style="font-weight:900; color:var(--accent)">${l.score}</div>
                </div>
            `).join('');
        }
    }

    window.onload = renderHistory;
</script>
