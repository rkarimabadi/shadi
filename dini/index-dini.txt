    const lessons = [
        { id: 1, title: "بینای توانا", icon: "ph-eye" },
        { id: 2, title: "استعانت از خدا", icon: "ph-hands-praying" },
        { id: 3, title: "تلخ یا شیرین", icon: "ph-info" },
        { id: 4, title: "عبور آسان", icon: "ph-path" },
        { id: 5, title: "پیامبر رحمت", icon: "ph-heart" },
        { id: 6, title: "اسوه فداکاری و عدالت", icon: "ph-shield-check" },
        { id: 7, title: "برترین بانو", icon: "ph-star" },
        { id: 8, title: "افتخار بندگی", icon: "ph-medal" },
        { id: 9, title: "به سوی پاکی", icon: "ph-drop" },
        { id: 10, title: "ستون دین", icon: "ph-mosque" },
        { id: 11, title: "نماز جماعت", icon: "ph-users-three" },
        { id: 12, title: "نشان ارزشمندی", icon: "ph-crown" },
        { id: 13, title: "بر بال فرشتگان", icon: "ph-bird" },
        { id: 14, title: "کمال همنشین", icon: "ph-user-plus" },
        { id: 15, title: "مزدوران شیطان", icon: "ph-warning-octagon" }
    ];

    const baseUrl = "https://rkarimabadi.github.io/shadi/dini/";

    function renderLessons(data) {
        const grid = document.getElementById('lessonsGrid');
        grid.innerHTML = data.map(lesson => `
            <a href="${baseUrl}lesson-${lesson.id}.html" class="lesson-card">
                <div class="icon-wrapper">
                    <i class="ph ${lesson.icon}"></i>
                </div>
                <div class="lesson-info">
                    <h3>درس ${lesson.id}</h3>
                    <p>${lesson.title}</p>
                </div>
                <i class="ph ph-caret-left" style="margin-right: auto; color: #cbd5e1;"></i>
            </a>
        `).join('');
    }

    function filterLessons() {
        const term = document.getElementById('searchInput').value.toLowerCase();
        const filtered = lessons.filter(l => 
            l.title.toLowerCase().includes(term) || 
            l.id.toString().includes(term)
        );
        renderLessons(filtered);
    }

    // Initial render
    window.onload = () => renderLessons(lessons);
</script>
